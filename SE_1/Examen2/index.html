void cambio_modo(uint gpio, uint32_t events) {

    last_time = now;                  // Actualiza el tiempo de √∫ltima activaci√≥n v√°lida

    // Identifica qu√© bot√≥n caus√≥ la interrupci√≥n y activa su bandera

void configurar_servo() {
    // Configura el pin como salida PWM
    gpio_set_function(SERVO_PIN, GPIO_FUNC_PWM);

    // Obtiene el n√∫mero de slice PWM asignado al pin
    slice_num = pwm_gpio_to_slice_num(SERVO_PIN);

    // Divisor de clock para ticks de 1¬µs (125MHz/125 = 1MHz)
    pwm_set_clkdiv(slice_num, 125.0f);

    // Configura periodo de 20ms (20000¬µs) para frecuencia de 50Hz
    pwm_set_wrap(slice_num, 20000);

    // Habilita la salida PWM
    pwm_set_enabled(slice_num, true);
}

// Funci√≥n para mover el servo a un √°ngulo espec√≠fico
void mover_servo(int angulo) {

    if (angulo < 0) angulo = 0;        // No permitir √°ngulos negativos
    if (angulo > 180) angulo = 180;    // No permitir √°ngulos mayores a 180

    // Convierte √°ngulo a ancho de pulso:
    // - M√≠nimo: 500¬µs (0 grados)
    // - M√°ximo: 2500¬µs (180 grados)
    int pulso_us = 500 + (angulo * 2000) / 180;

    // Aplica el ancho de pulso al PWM
    pwm_set_gpio_level(SERVO_PIN, pulso_us);

    // Muestra el √°ngulo actual por consola
    printf("Servo en %d¬∞\n", angulo);
}

void modo1() {

        // Agrega el car√°cter al comando en construcci√≥n
        comando += c;

    // === Comando BORRAR: Elimina todas las posiciones guardadas ===

        return;                         // Sale de la funci√≥n modo1

    // === Comando ESCRIBIR: Ingresa una nueva secuencia de posiciones ===
    if (comando == "escribir") {
        // Solicita al usuario ingresar los valores con el formato correcto
        printf("Ingrese valores (0‚Äì180) separados por comas, termine con ';'\n> ");
        cantidad = 0;                    // Reinicia el contador para la nueva lista
        string valor = "";              // Buffer temporal para construir cada n√∫mero
        while (true) {
            // Verifica si se debe salir del modo
            if (boton_modo_presionado) return;
            // Intenta leer un car√°cter de forma no bloqueante
            int ch = getchar_timeout_us(0);
            // Si no hay datos disponibles, contin√∫a el loop
            if (ch == PICO_ERROR_TIMEOUT) continue;
            // Convierte el c√≥digo ASCII a car√°cter

            // Si encuentra separador y hay un n√∫mero acumulado

                // Valida que el √°ngulo est√© en el rango permitido
                if (num < 0 || num > 180) {

                    return;
                }
                // Almacena el n√∫mero y aumenta el contador
                lista[cantidad++] = num;
                valor = "";              // Limpia el buffer para el siguiente n√∫mero
                if (c == ';') break;    // Si es punto y coma, termina la entrada
            } else if (c != ',' && c != ' ') {
                // Si no es separador ni espacio, acumula el d√≠gito
                valor += c;
            }

            // Termina si encuentra punto y coma (redundante pero seguro)
            if (c == ';') break;

        // Verifica que se haya ingresado al menos un valor
        if (cantidad == 0) {
            printf("Error argumento invalido\n");
        } else {
            // Muestra la lista completa ingresada

            for (int i = 0; i < cantidad; i++) {
                printf("%d", lista[i]);
                // Agrega coma excepto despu√©s del √∫ltimo n√∫mero
                if (i < cantidad - 1) printf(", ");
            }
            printf("\n");
        }
        return;
    }

    // === Comando REEMPLAZAR: Modifica una posici√≥n existente ===
    if (comando == "reemplazar") {
        // Verifica que haya posiciones para modificar
        if (cantidad == 0) {
            printf("No hay valores en la lista.\n");
            return;
        }

        // Solicita el √≠ndice y nuevo valor con formato espec√≠fico
        printf("Ingrese √≠ndice y nuevo valor separados por coma, termine con ';'\n> ");

                valor = "";                          // Limpia buffer para siguiente n√∫mero

                // Acumula d√≠gitos si no es separador ni espacio

            // Sale si encuentra punto y coma (redundante pero seguro)
            if (c == ';') break;

        // Verifica que se ingresaron dos n√∫meros

            return;
        }

        // Procesa los valores ingresados

        // Valida que el √≠ndice est√© en rango

            return;

            // Agrega coma excepto despu√©s del √∫ltimo n√∫mero
            if (j < cantidad - 1) printf(", ");

        printf("\n");
        return;
    }

void modo2() {

    if (cantidad == 0) {

        sleep_ms(500);

    printf("Moviendo continuamente seg√∫n lista...\n");
    while (!boton_modo_presionado) {           // Loop principal

void modo3() {
    // Verificar que haya posiciones para mostrar
    if (cantidad == 0) {
        printf("No hay valores en la lista.\n");
        sleep_ms(500);
        return;
    }

    static int indice = 0;                    // Mantener posici√≥n entre llamadas
    mover_servo(lista[indice]);               // Mostrar posici√≥n inicial

    while (!boton_modo_presionado) {

        if (boton_next_presionado) {
            boton_next_presionado = false;    // Limpiar bandera
            indice++;                         // Avanzar
            if (indice >= cantidad) indice = 0; // Volver al inicio si necesario
            mover_servo(lista[indice]);       // Actualizar posici√≥n
        }

        // Procesar bot√≥n PREV
        if (boton_prev_presionado) {
            boton_prev_presionado = false;    // Limpiar bandera
            indice--;                         // Retroceder
            if (indice < 0) indice = cantidad - 1; // Ir al final si necesario
            mover_servo(lista[indice]);       // Actualizar posici√≥n
        }

        sleep_ms(20);                        // Breve delay para estabilidad
    }
}


int main() {
    // Inicializaci√≥n de comunicaci√≥n serial

    sleep_ms(500);                       // Espera para estabilizaci√≥n del sistema

    // Bot√≥n de cambio de modo

    // Bot√≥n para avanzar en la secuencia
    gpio_init(BOTON_NEXT);               // Inicializa el pin
    gpio_set_dir(BOTON_NEXT, GPIO_IN);   // Configura como entrada

        // Manejo del cambio de modo

            boton_modo_presionado = false;    // Limpia la bandera

        // Ejecuci√≥n del modo actual
        if (modo == 1) modo1();              // Modo Entrenamiento

        sleep_ms(50);

}
